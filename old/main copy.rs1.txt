use std::sync::{Arc, Mutex, RwLock};
use std::thread;

/// Структура для параллельного доступа к элементам вектора
/// Сочетает возможность изменения размера с минимальной блокировкой при доступе к элементам
struct ParallelVector<T> {
    // Внутренний вектор Arc<Mutex<T>> для индивидуальной блокировки элементов
    data: RwLock<Vec<Arc<Mutex<T>>>>,
}

impl<T> ParallelVector<T> {
    /// Создает новый пустой вектор
    fn new() -> Self {
        ParallelVector {
            data: RwLock::new(Vec::new()),
        }
    }

    /// Создает вектор заданного размера с значениями по умолчанию
    fn with_capacity(size: usize, default_value: T) -> Self
    where
        T: Clone,
    {
        let mut vec = Vec::with_capacity(size);
        for _ in 0..size {
            vec.push(Arc::new(Mutex::new(default_value.clone())));
        }

        ParallelVector {
            data: RwLock::new(vec),
        }
    }

    /// Возвращает количество элементов в векторе
    fn len(&self) -> usize {
        let data_guard = self.data.read().unwrap(); // 🔓 Множественное чтение
        data_guard.len()
    }

    /// Доступ к элементу по индексу с возможностью модификации
    fn get(&self, index: usize) -> Option<Arc<Mutex<T>>> {
        let data_guard = self.data.read().unwrap(); // 🔓 Множественное чтение
        data_guard.get(index).cloned() // ✅ Возвращаем клон Arc
    }

    /// Изменяет размер вектора (редкая операция)
    fn resize(&self, new_size: usize, default_value: T)
    where
        T: Clone,
    {
        let mut data_guard = self.data.write().unwrap(); // 🔒 Эксклюзивная запись
        let current_len = data_guard.len(); // Сохраняем длину для избежания повторных заимствований

        //println!("{}", new_size);
        if new_size > current_len {
            // Увеличиваем размер - добавляем новые элементы
            data_guard.reserve(new_size - current_len);
            for _ in 0..(new_size - current_len) {
                data_guard.push(Arc::new(Mutex::new(default_value.clone())));
            }
        } else if new_size < current_len {
            // Уменьшаем размер - удаляем лишние элементы
            // Извлекаем удаляемые элементы для возможного освобождения памяти
            let removed_elements: Vec<Arc<Mutex<T>>> = data_guard.drain(new_size..).collect();

            // Явно освобождаем блокировку перед потенциально долгими операциями
            drop(data_guard);

            // Пытаемся освободить память удаленных элементов
            for arc in removed_elements {
                if let Ok(mutex) = Arc::try_unwrap(arc) {
                    let _ = mutex.into_inner().ok(); // Игнорируем ошибки
                }
                // Если другие ссылки есть - память освободится автоматически
            }
        }
    }

    /// Добавляет элемент в конец вектора
    fn push(&self, value: T) {
        let mut data_guard = self.data.write().unwrap(); // 🔒 Эксклюзивная запись
        data_guard.push(Arc::new(Mutex::new(value)));
    }

    /// Удаляет и возвращает последний элемент
    fn pop(&self) -> Option<Arc<Mutex<T>>> {
        let mut data_guard = self.data.write().unwrap(); // 🔒 Эксклюзивная запись
        data_guard.pop()
    }

    /// Выполняет операцию над элементом с автоматической блокировкой
    fn with_element<F, R>(&self, index: usize, f: F) -> Option<R>
    where
        F: FnOnce(&mut T) -> R,
    {
        self.get(index).map(|arc| {
            let mut guard = arc.lock().unwrap(); // 🔒 Блокировка элемента
            f(&mut *guard)
        })
    }
}

// Пример использования
fn main() {
    // Создаем вектор из 1000 элементов со значением 0
    let parallel_vec = Arc::new(ParallelVector::with_capacity(1000, 0));

    let mut handles = vec![];

    // Запускаем 10 потоков для работы с вектором
    for thread_id in 0..10 {
        let vec_clone = Arc::clone(&parallel_vec);

        let handle = thread::spawn(move || {
            // Каждый поток работает со своим набором элементов
            for i in (thread_id * 100)..((thread_id + 1) * 100) {
                if i < vec_clone.len() {
                    // Модифицируем элемент
                    vec_clone.with_element(i, |value| {
                        *value += i; // Пример операции
                    });
                }
            }

            // Иногда поток может изменить размер вектора (с низкой вероятностью)
            if thread_id == 0 && rand::random::<f32>() < 0.01 {
                println!("Поток {} изменяет размер вектора", thread_id);
                vec_clone.resize(1500, 0);
            }
        });

        handles.push(handle);
    }

    // Главный поток также может работать с вектором
    for i in 0..50 {
        parallel_vec.with_element(i, |value| {
            *value *= 2; // Удваиваем значение
        });
    }

    // Ожидаем завершения всех потоков
    for handle in handles {
        handle.join().unwrap();
    }

    // Выводим статистику
    println!("Размер вектора: {}", parallel_vec.len());

    // Пример чтения некоторых значений
    for i in (0..10).step_by(100) {
        if let Some(value) = parallel_vec.get(i) {
            let guard = value.lock().unwrap();
            println!("vec[{}] = {}", i, *guard);
        }
    }
}

#[cfg(test)]
mod tests2 {
    use super::*;

    #[test]
    fn test_parallel_access() {
        let vec = Arc::new(ParallelVector::with_capacity(10, 0));
        let mut handles = vec![];

        // Тестируем параллельный доступ
        for i in 0..5 {
            let vec_clone = Arc::clone(&vec);
            handles.push(thread::spawn(move || {
                for j in 0..10 {
                    vec_clone.with_element(j, |value| {
                        *value += 1;
                    });
                }
            }));
        }

        for handle in handles {
            handle.join().unwrap();
        }

        // Проверяем, что все элементы были обновлены
        for i in 0..10 {
            let value = vec.get(i).unwrap();
            let guard = value.lock().unwrap();
            assert_eq!(*guard, 5); // Каждый элемент увеличен 5 раз
        }
    }

    #[test]
    fn test_resize() {
        let vec = ParallelVector::new();
        vec.resize(5, 0);
        assert_eq!(vec.len(), 5);

        vec.resize(3, 0);
        assert_eq!(vec.len(), 3);

        vec.resize(10, 42);
        assert_eq!(vec.len(), 10);

        // Проверяем значения новых элементов
        for i in 3..10 {
            let value = vec.get(i).unwrap();
            let guard = value.lock().unwrap();
            assert_eq!(*guard, 42);
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use std::sync::atomic::{AtomicU64, Ordering};
    use std::time::Instant;

    // Простой линейный конгруэнтный генератор для детерминистического тестирования
    struct SimpleRng {
        state: u64,
    }

    impl SimpleRng {
        fn new(seed: u64) -> Self {
            Self { state: seed }
        }

        fn next(&mut self) -> u64 {
            self.state = self
                .state
                .wrapping_mul(6364136223846793005)
                .wrapping_add(1442695040888963407);
            self.state
        }

        fn gen_range(&mut self, low: usize, high: usize) -> usize {
            let range = (high - low) as u64;
            (low as u64 + self.next() % range) as usize
        }

        fn gen_prob(&mut self) -> f64 {
            (self.next() % 10000) as f64 / 10000.0
        }
    }

    #[test]
    fn performance_test() {
        const NUM_THREADS: usize = 10_00;
        const VECTOR_SIZE: usize = 5_00;
        const OPS_PER_THREAD: usize = 20_000;
        const RESIZE_PROBABILITY: f64 = 0.000001;
        {
            println!("Starting performance test...");
            println!("Threads: {}", NUM_THREADS);
            println!("Vector size: {}", VECTOR_SIZE);
            println!("Operations per thread: {}", OPS_PER_THREAD);
            println!("Resize probability: {}", RESIZE_PROBABILITY);
        }
        // Создаем вектор и заполняем его нулями
        //let vector = Arc::new(ParallelVector::with_capacity(VECTOR_SIZE, 0u64));

        let mut vec_vec_vec = Vec::with_capacity(1000);

        for _ in 0..10 {
            vec_vec_vec.push(Arc::new(ParallelVector::with_capacity(VECTOR_SIZE, 0u64)));
        }

        let start = Instant::now();

        let mut handles = Vec::with_capacity(NUM_THREADS);

        for thread_id in 0..NUM_THREADS {
            //let vector_clone = Arc::clone(&vector);
            let vector_clone = Arc::clone(&vec_vec_vec[thread_id % vec_vec_vec.len()]);
            let total_ops_clone = Arc::new(AtomicU64::new(0));

            let handle = thread::spawn(move || {
                let mut rng = SimpleRng::new((thread_id + 1) as u64);
                let mut local_ops = 0;

                for _ in 0..OPS_PER_THREAD {
                    //thread::sleep(std::time::Duration::from_micros(10));
                    // С вероятностью RESIZE_PROBABILITY изменяем размер вектора
                    if rng.gen_prob() < RESIZE_PROBABILITY {
                        let current_size = vector_clone.len();
                        let new_size = if rng.gen_prob() < 0.5 {
                            current_size.saturating_sub(rng.gen_range(1, 1000))
                        } else {
                            current_size + rng.gen_range(1, 1000)
                        };
                        vector_clone.resize(new_size, 0);
                    } else {
                        // Выбираем случайный элемент для чтения/записи
                        let current_size = vector_clone.len();
                        if current_size > 0 {
                            let index = rng.gen_range(0, current_size);

                            // Читаем и модифицируем элемент
                            //thread::sleep(std::time::Duration::from_micros(10));
                            vector_clone.with_element(index, |value| {
                                thread::sleep(std::time::Duration::from_micros(10));
                                *value = value.wrapping_add(1);
                            });
                        }
                    }

                    local_ops += 1;
                }

                total_ops_clone.store(local_ops, Ordering::Relaxed);
            });

            handles.push(handle);
        }

        // Ждем завершения всех потоков и собираем результаты

        for handle in handles {
            handle.join().unwrap();
            // В реальном коде здесь нужно было бы аккумулировать результаты из total_ops_clone
        }

        let duration = start.elapsed();
        let total_operations = NUM_THREADS * OPS_PER_THREAD;
        let ops_per_second = total_operations as f64 / duration.as_secs_f64();

        println!("Test completed!");
        println!("Time taken: {:?}", duration);
        println!("Total operations: {}", total_operations);
        println!("Operations per second: {:.2}", ops_per_second);
        println!(
            "Operations per second: {:.2}M",
            ops_per_second / 1_000_000.0
        );
        assert_ne!(1, 1,);
        //println!("Final vector size: {}", vector.len());

        // Проверяем целостность данных - сумма всех изменений должна быть равна количеству операций записи
        let mut total_sum = 0u64;
        //for i in 0..vector.len() {
        //    if let Some(value) = vector.get(i) {
        //        let guard = value.lock().unwrap();
        //        total_sum = total_sum.wrapping_add(*guard);
        //    }
        //}

        println!("Total sum of all elements: {}", total_sum);

        // Убеждаемся, что тест выполнился достаточно быстро
        assert!(
            duration.as_secs() < 300, // 5 минут - разумный лимит для такого теста
            "Test took too long: {:?}",
            duration
        );

        // Выводим предупреждение, если производительность низкая
        if ops_per_second < 1_000_000.0 {
            println!(
                "WARNING: Low performance detected: {:.2} ops/s",
                ops_per_second
            );
        }
    }
