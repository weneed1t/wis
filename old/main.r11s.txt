use rand;
use std::alloc::{alloc, dealloc, Layout};
use std::sync::{
    atomic::{AtomicPtr, AtomicUsize},
    Arc, Mutex,
};
use std::thread;
/// Небезопасная структура с преднамеренными гонками данных
/// ⚠️ ПРЕДУПРЕЖДЕНИЕ: Этот код создает неопределенное поведение!
/// Используется только для изучения гонок данных и не должен использоваться в продакшене.
struct UnsafeArray {
    ptr: *mut u32,
    len: usize,
    block: Mutex<u32>,
}

struct MadnessArr<T: Clone + Copy> {
    data: AtomicPtr<Box<[Arc<Mutex<Vec<Option<(T, u64)>>>>]>>,
    elems: AtomicUsize,
    block: Mutex<u32>,
}

impl<T: Copy + Clone> MadnessArr<T> {
    pub fn new(lensiz: usize, coof_full: f64) -> Result<Self, &'static str> {
        if !((0.3..=0.8).contains(&coof_full)) {
            return Err("The coof_full value should be between 0.3 and 0.8, inclusive");
        }
        if lensiz == 0 || lensiz > u64::MAX as usize {
            return Err("lensiz == 0 || lensiz > u64::MAX as usize");
        }

        let data: Vec<Arc<Mutex<Vec<Option<(T, u64)>>>>> = (0..lensiz)
            .map(|_| Arc::new(Mutex::new(Vec::new())))
            .collect();

        unsafe {
            Ok(MadnessArr {
                data: AtomicPtr::new(data.into_boxed_slice().as_ptr()),
                elems: AtomicUsize::new(0),
                block: Mutex::new(0),
            })
        }
    }

    pub fn insert(id: u64) -> Result<(), &'static str> {
        Err("()")
    }

    pub fn remove() -> Result<(), &'static str> {
        Err("()")
    }

    pub fn get() -> Result<(), &'static str> {
        Err("()")
    }
}

// Вручную реализуем Send и Sync для демонстрации небезопасного поведения
unsafe impl Send for UnsafeArray {}
unsafe impl Sync for UnsafeArray {}

impl UnsafeArray {
    /// Создает новый массив заданного размера
    fn new(size: usize) -> Self {
        let layout = Layout::array::<u32>(size).expect("Invalid layout");
        let ptr = unsafe { alloc(layout) as *mut u32 };

        // Инициализируем память нулями
        unsafe {
            for i in 0..size {
                ptr.add(i).write(0);
            }
        }

        UnsafeArray {
            ptr,
            len: size,
            block: Mutex::new(100),
        }
    }

    /// Возвращает значение по индексу (небезопасно, без проверок)
    fn get_item(&self, index: usize) -> u32 {
        let mut a = self.block.lock().unwrap();
        unsafe {
            // Чтение без синхронизации - возможна гонка данных
            let res = *self.ptr.add(index);

            *a = a.wrapping_add(res);
            return res;
        }
    }

    /// Изменяет значение по индексу (небезопасно, без синхронизации)
    fn edit_item(&self, index: usize, value: u32) {
        let mut a = self.block.lock().unwrap();
        unsafe {
            // Запись без синхронизации - гарантированная гонка данных
            *self.ptr.add(index) = value;

            *a = a.wrapping_add(index as u32);
        }
    }

    /// Возвращает длину массива
    fn get_len(&self) -> usize {
        self.len
    }

    /// Специальный метод для демонстрации гонки данных
    /// Увеличивает значение на 1 без атомарных операций
    fn increment_unsafe(&self, index: usize) {
        //let mut a = self.block.lock().unwrap();
        unsafe {
            // Чтение-изменение-запись без синхронизации
            let current = *self.ptr.add(index);
            *self.ptr.add(index) = current + 1;
            //*a = a.wrapping_add(index as u32);
        }
    }
}

impl Drop for UnsafeArray {
    fn drop(&mut self) {
        // Освобождаем память при удалении
        if !self.ptr.is_null() {
            let layout = Layout::array::<u32>(self.len).expect("Invalid layout");
            unsafe {
                dealloc(self.ptr as *mut u8, layout);
            }
        }
    }
}

// Пример использования для демонстрации гонок данных
fn main() {
    // Создаем небезопасный массив и оборачиваем в Arc для многопоточного доступа
    let unsafe_array = Arc::new(UnsafeArray::new(10));

    println!("Исходное состояние массива:");
    for i in 0..unsafe_array.get_len() {
        println!("array[{}] = {}", i, unsafe_array.get_item(i));
    }

    // Создаем несколько потоков, которые будут одновременно изменять данные
    let mut handles = vec![];

    for thread_id in 0..5 {
        let array_clone = Arc::clone(&unsafe_array);

        let handle = thread::spawn(move || {
            // Каждый поток пытается увеличить значения в массиве
            for i in 0..array_clone.get_len() {
                for _ in 0..1000 {
                    array_clone.increment_unsafe(i);
                }
            }
            println!("Поток {} завершил работу", thread_id);
        });

        handles.push(handle);
    }

    // Ждем завершения всех потоков
    for handle in handles {
        handle.join().unwrap();
    }

    println!("\nСостояние массива после гонки данных:");
    for i in 0..unsafe_array.get_len() {
        println!("array[{}] = {}", i, unsafe_array.get_item(i));
    }

    // Ожидаемый результат: 5000 для каждого элемента (5 потоков × 1000 инкрементов)
    // Фактический результат: меньше 5000 из-за гонки данных
    println!("\nОжидаемая сумма каждого элемента: 5000");
    println!(
        "Реальная сумма каждого элемента: {} (гонки данных!)",
        unsafe_array.get_item(0)
    );
}
