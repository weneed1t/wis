/// Высокопроизводительный параллельный вектор с минимальными блокировками
struct ParallelVector<T> {
    data: ParkingRwLock<Vec<ParkingMutex<T>>>,
}

impl<T> ParallelVector<T> {
    /// Создает новый пустой вектор
    fn new() -> Self {
        ParallelVector {
            data: ParkingRwLock::new(Vec::new()),
        }
    }

    /// Создает вектор заданного размера с значениями по умолчанию
    fn with_capacity(size: usize, default_value: T) -> Self
    where
        T: Clone,
    {
        let mut vec = Vec::with_capacity(size);
        for _ in 0..size {
            vec.push(ParkingMutex::new(default_value.clone()));
        }

        ParallelVector {
            data: ParkingRwLock::new(vec),
        }
    }

    /// Возвращает количество элементов в векторе
    fn len(&self) -> usize {
        let data_guard = self.data.read();
        data_guard.len()
    }

    /// Выполняет операцию над элементом с автоматической блокировкой
    fn with_element<F, R>(&self, index: usize, f: F) -> Option<R>
    where
        F: FnOnce(&mut T) -> R,
    {
        let data_guard = self.data.read();
        let mutex = data_guard.get(index)?;
        let mut guard = mutex.lock();
        Some(f(&mut *guard))
    }
    //старые нейроботы пытаются отличить новый нейроботов от старых нейробоотв с другими аками от школьников тренерующих новый нейроботов
    /// Изменяет размер вектора (редкая операция)
    fn resize(&self, new_size: usize, default_value: T)
    where
        T: Clone,
    {
        let mut data_guard = self.data.write();
        let current_len = data_guard.len();

        if new_size > current_len {
            // Увеличиваем размер - добавляем новые элементы
            data_guard.reserve(new_size - current_len);
            for _ in 0..(new_size - current_len) {
                data_guard.push(ParkingMutex::new(default_value.clone()));
            }
        } else {
            // Уменьшаем размер - удаляем лишние элементы
            data_guard.truncate(new_size);
        }
    }

    /// Добавляет элемент в конец вектора
    fn push(&self, value: T) {
        let mut data_guard = self.data.write();
        data_guard.push(ParkingMutex::new(value));
    }

    /// Удаляет и возвращает последний элемент
    fn pop(&self) -> Option<T>
    where
        T: Default,
    {
        let mut data_guard = self.data.write();
        data_guard.pop().map(|mutex| {
            let guard = mutex.into_inner();
            guard
        })
    }
}

// Пример использования
fn main() {
    // Создаем вектор из 1000 элементов со значением 0
    let parallel_vec = Arc::new(ParallelVector::with_capacity(1000, 0));

    let mut handles = vec![];

    // Запускаем 10 потоков для работы с вектором
    for thread_id in 0..10 {
        let vec_clone = Arc::clone(&parallel_vec);

        let handle = thread::spawn(move || {
            // Каждый поток работает со своим набором элементов
            for i in (thread_id * 100)..((thread_id + 1) * 100) {
                if i < 1000 {
                    // Проверяем границы
                    // Модифицируем элемент
                    vec_clone.with_element(i, |value| {
                        *value += i; // Пример операции
                    });
                }
            }

            // Иногда поток может изменить размер вектора (с низкой вероятностью)
            //if thread_id == 0 && rand::thread_rng().gen::<f32>() < 0.01 {
            //    println!("Поток {} изменяет размер вектора", thread_id);
            //    vec_clone.resize(1500, 0);
            // }
        });

        handles.push(handle);
    }

    // Главный поток также может работать с вектором
    for i in 0..50 {
        parallel_vec.with_element(i, |value| {
            *value *= 2; // Удваиваем значение
        });
    }

    // Ожидаем завершения всех потоков
    for handle in handles {
        handle.join().unwrap();
    }

    // Выводим статистику
    println!("Размер вектора: {}", parallel_vec.len());

    // Пример чтения некоторых значений
    for i in (0..10).step_by(2) {
        parallel_vec.with_element(i, |value| {
            println!("vec[{}] = {}", i, *value);
        });
    }
}

    // Простой линейный конгруэнтный генератор для детерминистического тестирования
    struct SimpleRng {
        state: u64,
    }

    impl SimpleRng {
        fn new(seed: u64) -> Self {
            Self { state: seed }
        }

        fn next(&mut self) -> u64 {
            self.state = self
                .state
                .wrapping_mul(6364136223846793005)
                .wrapping_add(1442695040888963407);
            self.state
        }

        fn gen_range(&mut self, low: usize, high: usize) -> usize {
            (self.next() as usize % (high - low)) + low
        }

        fn gen_prob(&mut self) -> f64 {
            (self.next() % 10000) as f64 / 10000.0
        }
    }

    #[test]
    fn test_parallel_access() {
        let vec = Arc::new(ParallelVector::with_capacity(10, 0));
        let mut handles = vec![];

        // Тестируем параллельный доступ
        for i in 0..5 {
            let vec_clone = Arc::clone(&vec);
            handles.push(thread::spawn(move || {
                for j in 0..10 {
                    vec_clone.with_element(j, |value| {
                        *value += 1;
                    });
                }
            }));
        }

        for handle in handles {
            handle.join().unwrap();
        }

        // Проверяем, что все элементы были обновлены
        for i in 0..10 {
            vec.with_element(i, |value| {
                assert_eq!(*value, 5); // Каждый элемент увеличен 5 раз
            });
        }
    }

        let vec = ParallelVector::new();

        vec.resize(10, 42);
        assert_eq!(vec.len(), 10);

use std::collections::HashMap;
use std::hint::black_box;
use std::time::Instant;

/// Простой линейный конгруэнтный генератор для детерминистичности теста
struct SimpleRng {
    state: u64,
}

impl SimpleRng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next(&mut self) -> u64 {
        self.state = self
            .state
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1442695040888963407);
        self.state
    }

    fn gen_range(&mut self, low: u64, high: u64) -> u64 {
        (self.next() % (high - low)) + low
    }
}

#[test]
fn hashmap_performance_test() {
    const NUM_OPERATIONS: usize = 1_000_000;
    const KEY_RANGE: u64 = 1_000_000;

    println!("Starting HashMap performance test...");
    println!("Operations: {}", NUM_OPERATIONS);
    println!("Key range: {}", KEY_RANGE);

    // Создаем и заполняем HashMap
    let mut map = HashMap::new();
    let mut rng = SimpleRng::new(123456789);

    // Предварительное заполнение
    for _ in 0..KEY_RANGE {
        let key = rng.gen_range(0, KEY_RANGE);
        let value = rng.next();
        map.insert(key, value);
    }

    // Тестируем операции доступа
    let mut rng = SimpleRng::new(987654321);

    // Тест операции GET
    let start = Instant::now();
    for _ in 0..NUM_OPERATIONS {
        let key = rng.gen_range(0, KEY_RANGE);
        black_box(map.get(&key));
    }
    let get_duration = start.elapsed();
    let get_ops_per_sec = NUM_OPERATIONS as f64 / get_duration.as_secs_f64();

    // Тест операции INSERT (с перезаписью)
    let start = Instant::now();
    for _ in 0..NUM_OPERATIONS {
        let key = rng.gen_range(0, KEY_RANGE);
        let value = rng.next();
        black_box(map.insert(key, value));
    }
    let insert_duration = start.elapsed();
    let insert_ops_per_sec = NUM_OPERATIONS as f64 / insert_duration.as_secs_f64();

    // Тест операции CONTAINS (проверка наличия ключа)
    let start = Instant::now();
    for _ in 0..NUM_OPERATIONS {
        let key = rng.gen_range(0, KEY_RANGE);
        black_box(map.contains_key(&key));
    }
    let contains_duration = start.elapsed();
    let contains_ops_per_sec = NUM_OPERATIONS as f64 / contains_duration.as_secs_f64();

    // Тест операции ENTRY (получение или вставка)
    let start = Instant::now();
    for _ in 0..NUM_OPERATIONS {
        let key = rng.gen_range(0, KEY_RANGE);
        black_box(map.entry(key).or_insert(rng.next()));
    }
    let entry_duration = start.elapsed();
    let entry_ops_per_sec = NUM_OPERATIONS as f64 / entry_duration.as_secs_f64();

    println!("Test completed!");
    println!("GET operations: {:.2} ops/s", get_ops_per_sec);
    println!("INSERT operations: {:.2} ops/s", insert_ops_per_sec);
    println!("CONTAINS operations: {:.2} ops/s", contains_ops_per_sec);
    println!("ENTRY operations: {:.2} ops/s", entry_ops_per_sec);
    println!("Final map size: {}", map.len());

    // Проверяем, что производительность адекватна
    assert!(
        get_ops_per_sec > 1_000_000.0,
        "GET performance too low: {:.2} ops/s",
        get_ops_per_sec
    );

    assert!(
        insert_ops_per_sec > 1_000_000.0,
        "INSERT performance too low: {:.2} ops/s",
        insert_ops_per_sec
    );
}
