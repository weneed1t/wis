use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Instant;
// Структура для безопасного доступа к элементам массива
struct ConcurrentArray {
    data: Mutex<Vec<Arc<Mutex<u32>>>>,
}

impl ConcurrentArray {
    // Создание нового массива заданного размера
    fn new(size: usize) -> Self {
        let mut vec = Vec::with_capacity(size);
        for _ in 0..size {
            vec.push(Arc::new(Mutex::new(0)));
        }

        ConcurrentArray {
            data: Mutex::new(vec),
        }
    }

    // Получение доступа к элементу по индексу
    fn get_element(&self, index: usize) -> Option<Arc<Mutex<u32>>> {
        let data_guard = self.data.lock().unwrap();
        data_guard.get(index).cloned()
    }

    // Изменение размера массива
    fn resize_o(&self, new_size: usize) {
        let mut data_guard = self.data.lock().unwrap();
        if new_size > data_guard.len() {
            // Увеличиваем массив
            for _ in data_guard.len()..new_size {
                data_guard.push(Arc::new(Mutex::new(0)));
            }
        } else {
            // Уменьшаем массив
            data_guard.truncate(new_size);
        }
    }

    fn resize(&self, new_size: usize) {
        // Первый этап: удаление элементов из вектора под блокировкой
        let removed_elements = {
            let mut data_guard = self.data.lock().unwrap();
            let current_size = data_guard.len();

            if new_size > current_size {
                // Увеличиваем массив
                data_guard.reserve(new_size - current_size);
                for _ in 0..(new_size - current_size) {
                    data_guard.push(Arc::new(Mutex::new(0)));
                }
                return; // Нет элементов для удаления
            } else if new_size < current_size {
                // Уменьшаем массив - извлекаем удаляемые элементы
                data_guard.drain(new_size..).collect::<Vec<_>>()
            } else {
                return; // Размер не изменился
            }
        };

        // Второй этап: освобождение памяти без блокировки основного вектора
        for arc in removed_elements {
            // Пытаемся преобразовать Arc в единоличное владение
            if let Ok(mutex) = Arc::try_unwrap(arc) {
                // Если других ссылок нет - освобождаем память
                let _ = mutex.into_inner().unwrap();
            }
            // Если другие ссылки есть - память освободится автоматически
            // когда последняя ссылка будет удалена
        }
    }

    // Добавление обработки элемента с автоматической блокировкой
    fn process_element<F>(&self, index: usize, processor: F) -> Result<(), String>
    where
        F: FnOnce(&mut u32),
    {
        let element_arc = self
            .get_element(index)
            .ok_or_else(|| format!("Index {} out of bounds", index))?;

        let mut element_guard = element_arc.lock().unwrap();
        processor(&mut element_guard);

        Ok(())
    }
}

// Пример использования
fn main() {
    t2_test1();
    return;
    let array = Arc::new(ConcurrentArray::new(10));
    let mut handles = vec![];

    // Создаем несколько потоков для обработки элементов
    for i in 0..5 {
        let array_clone = Arc::clone(&array);
        let handle = thread::spawn(move || {
            // Обрабатываем элемент с индексом i
            for _ in 0..100 {
                array_clone
                    .process_element(i, |value| {
                        // Длительная обработка
                        *value += 1;
                        // Имитация обработки
                        thread::sleep(std::time::Duration::from_micros(100));
                    })
                    .unwrap();
            }
        });
        handles.push(handle);
    }

    // Пример изменения размера во время выполнения
    let array_clone = Arc::clone(&array);
    let resize_handle = thread::spawn(move || {
        thread::sleep(std::time::Duration::from_millis(10));
        array_clone.resize(15);
        println!("Массив увеличен до 15 элементов");
    });

    // Ждем завершения всех потоков
    for handle in handles {
        handle.join().unwrap();
    }
    resize_handle.join().unwrap();

    // Выводим результаты
    for i in 0..5 {
        if let Some(element) = array.get_element(i) {
            let value = element.lock().unwrap();
            println!("Element[{}] = {}", i, *value);
        }
    }
}

fn t2_test1() {
    let ops = 10_000_000;
    let mut a = 132u64;
    let mut b = 77u64;
    let mut c = 10u64;
    let mut ve = vec![1u64; 200_000_000];

    let start = Instant::now();
    for _ in 0..ops {
        a = ve[(ops * 213441) % ve.len()];
        b = ve[((ops * 213441) + 1) % ve.len()];
        for h in 0..5 {
            a = a.wrapping_add(b);
            a = a.rotate_left(7 * h);
            b ^= a;
        }
        c = c.wrapping_add(b);
        let ccc = ve.len();
        ve[(ops * 213441) % ccc] ^= c;
    }

    let r = start.elapsed();
    println!("hsh = {:.2}M\n{}", ops as f64 / r.as_micros() as f64, c);
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_concurrent_access() {
        let array = Arc::new(ConcurrentArray::new(5));
        let mut handles = vec![];

        for _ in 0..10 {
            let array_clone = Arc::clone(&array);
            handles.push(thread::spawn(move || {
                array_clone
                    .process_element(2, |value| {
                        *value += 1;
                    })
                    .unwrap();
            }));
        }

        for handle in handles {
            handle.join().unwrap();
        }

        let element = array.get_element(2).unwrap();
        let value = element.lock().unwrap();
        assert_eq!(*value, 10);
    }

    use std::sync::atomic::{AtomicUsize, Ordering};
    use std::time::Instant;

    // Простой линейный конгруэнтный генератор для детерминистического тестирования
    fn lcg(seed: &mut u32) -> u32 {
        //*seed = (*seed * 1103515245 + 12345) & 0x7fffffff;
        *seed = seed.wrapping_mul(1103515245).wrapping_add(12345) & 0x7fffffff;

        *seed
    }

    #[test]
    fn concurrent_array_performance_test() {
        const NUM_THREADS: usize = 100; // Количество потоков
        const INITIAL_SIZE: usize = 100; // Начальный размер массива
        const OPS_PER_THREAD: usize = 30000; // Операций на поток
        const RESIZE_PROBABILITY: f64 = 0.0; //0.0000001; // Вероятность изменения размера

        println!("Starting performance test...");
        println!("Threads: {}", NUM_THREADS);
        println!("Initial array size: {}", INITIAL_SIZE);
        println!("Operations per thread: {}", OPS_PER_THREAD);
        println!("Resize probability: {}", RESIZE_PROBABILITY);

        let mut vec_vec_vec_vec_vec = Vec::with_capacity(2000);

        for _ in 0..1000 {
            vec_vec_vec_vec_vec.push(Arc::new(ConcurrentArray::new(INITIAL_SIZE)));
        }

        let resize_count = AtomicUsize::new(0);

        let start = Instant::now();

        let mut handles = Vec::with_capacity(NUM_THREADS);

        for thread_id in 0..NUM_THREADS {
            //let array_clone = Arc::clone(&array);

            let array_clone =
                Arc::clone(&vec_vec_vec_vec_vec[thread_id % vec_vec_vec_vec_vec.len()]);

            let total_ops_clone = Arc::new(AtomicUsize::new(0));
            let resize_count_clone = Arc::new(AtomicUsize::new(0));

            let handle = thread::spawn(move || {
                //let mut seed = (thread_id + 1) as u32 * 123456789;
                let mut seed = thread_id.wrapping_add(1).wrapping_mul(123456789) as u32;
                seed = seed ^ seed.rotate_left(17);

                let mut local_ops = 0;
                let mut local_resizes = 0;

                for _ in 0..OPS_PER_THREAD {
                    // Получаем текущий размер массива
                    let current_size = {
                        let data_guard = array_clone.data.lock().unwrap();
                        data_guard.len()
                    };

                    if current_size == 0 {
                        continue;
                    }

                    // Генерируем случайный индекс
                    let rand_val = lcg(&mut seed);
                    let index = (rand_val as usize) % current_size;

                    // С вероятностью RESIZE_PROBABILITY изменяем размер
                    if (rand_val as f64 / u32::MAX as f64) < RESIZE_PROBABILITY {
                        let new_size = if rand_val % 2 == 0 {
                            current_size + 100
                        } else {
                            current_size.saturating_sub(100)
                        };

                        array_clone.resize(new_size);
                        local_resizes += 1;
                    }

                    // Обрабатываем элемент
                    if let Ok(()) = array_clone.process_element(index, |value| {
                        // Имитация обработки - просто инкрементируем значение
                        *value += 1;
                        // Небольшая задержка для имитации работы
                        thread::sleep(std::time::Duration::from_nanos(50000));

                        //thread::sleep(std::time::Duration::from_nanos(100_000_000));
                    }) {
                        local_ops += 1;
                    }
                }

                total_ops_clone.fetch_add(local_ops, Ordering::SeqCst);
                resize_count_clone.fetch_add(local_resizes, Ordering::SeqCst);
            });

            handles.push(handle);
        }

        // Ждем завершения всех потоков
        for handle in handles {
            handle.join().unwrap();
        }

        let duration = start.elapsed();
        let total_operations = NUM_THREADS * OPS_PER_THREAD;
        let ops_per_second = total_operations as f64 / duration.as_secs_f64();

        println!("Test completed!");
        println!("Time taken: {:?}", duration);
        println!("Total operations attempted: {}", total_operations);
        println!("Operations per second: {:.2}", ops_per_second);
        println!(
            "Operations per secondM : {:.2}M",
            ops_per_second / 1_000_000.0
        );
        println!("Resize operations: {}", resize_count.load(Ordering::SeqCst));

        // Проверяем целостность данных
        //let data_guard = array.data.lock().unwrap();
        //println!("Final array size: {}", data_guard.len());

        // Проверяем, что все элементы доступны и не повреждены
        //for (i, element) in data_guard.iter().enumerate() {
        //    let value = element.lock().unwrap();
        //    assert!(*value >= 0, "Element {} has negative value: {}", i, *value);
        //}

        // Убеждаемся, что тест выполнился достаточно быстро
        assert!(
            duration.as_secs() < 100,
            "Test took too long: {:?}",
            duration
        );
    }
